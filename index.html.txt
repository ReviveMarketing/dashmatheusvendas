<script>

let dados = JSON.parse(localStorage.getItem("dados")) || [];

function salvar() {

const entry = {
consultor: document.getElementById("consultor").value,
data: new Date().toISOString(),
stats: {
aguardando: +document.getElementById("aguardando").value || 0,
fechado: +document.getElementById("fechado").value || 0,
negociacao: +document.getElementById("negociacao").value || 0,
semDados: +document.getElementById("semDados").value || 0,
semPossibilidade: +document.getElementById("semPossibilidade").value || 0,
visita: +document.getElementById("visita").value || 0
}
};

dados.push(entry);
localStorage.setItem("dados", JSON.stringify(dados));

atualizar();
}

function calcularTotais() {

let total = 0;
let fechado = 0;
let negociacao = 0;

dados.forEach(d => {
Object.values(d.stats).forEach(v => total += v);
fechado += d.stats.fechado;
negociacao += d.stats.negociacao;
});

return {total, fechado, negociacao};
}

function ranking() {

let mapa = {};

dados.forEach(d => {
if(!mapa[d.consultor]) mapa[d.consultor] = 0;
mapa[d.consultor] += d.stats.fechado;
});

return Object.entries(mapa).sort((a,b)=>b[1]-a[1]);
}

function gerarInsight(totais) {
if(totais.fechado > totais.negociacao) {
return "Alta eficiência de fechamento.";
} else {
return "Pipeline maior que conversão — oportunidade de melhoria.";
}
}

let bar, linha;

function atualizar() {

const totais = calcularTotais();

document.getElementById("kpiFechado").innerText = totais.fechado;
document.getElementById("kpiTotal").innerText = totais.total;
document.getElementById("kpiConversao").innerText =
((totais.fechado/(totais.total||1))*100).toFixed(1) + "%";

document.getElementById("insight").innerText = gerarInsight(totais);

// ranking
const rank = ranking();
const div = document.getElementById("ranking");
div.innerHTML = "";

rank.forEach((r,i)=>{
div.innerHTML += `<div>#${i+1} ${r[0]} <b>${r[1]}</b></div>`;
});

// gráfico barras
const labels = ["Aguardando","Fechado","Negociação","Sem Dados","Sem Possibilidade","Visita"];
let valores = [0,0,0,0,0,0];

dados.forEach(d=>{
valores[0]+=d.stats.aguardando;
valores[1]+=d.stats.fechado;
valores[2]+=d.stats.negociacao;
valores[3]+=d.stats.semDados;
valores[4]+=d.stats.semPossibilidade;
valores[5]+=d.stats.visita;
});

if(bar) bar.destroy();
bar = new Chart(document.getElementById("graficoBar"), {
type: "bar",
data: { labels, datasets: [{ data: valores }] }
});

// linha
const timeline = dados.map(d=>({
data: new Date(d.data).toLocaleDateString(),
fechado: d.stats.fechado
}));

if(linha) linha.destroy();
linha = new Chart(document.getElementById("graficoLinha"), {
type: "line",
data: {
labels: timeline.map(t=>t.data),
datasets: [{ data: timeline.map(t=>t.fechado) }]
}
});

}

atualizar();
